Learn Concepts of AI Using Python

1 import json
2 import os
3 
4 KNOWLEDGE_BASE_FILE = os.path.join(os.getcwd(), "chatbot_knowledge.json")
5 
6 # Load or initialize the knowledge base
7 def load_knowledge():
8     if os.path.exists(KNOWLEDGE_BASE_FILE):
9         if os.path.getsize(KNOWLEDGE_BASE_FILE) > 0:
10             try:
11                 with open(KNOWLEDGE_BASE_FILE, "r") as file:
12                     data = json.load(file)
13                     print("[✓] Loaded existing knowledge base.")
14                     return data
15             except json.JSONDecodeError:
16                 print("[!] JSON file was corrupt. Starting fresh.")
17     print("[✓] No previous knowledge base found. Starting fresh.")
18     return {}
19 
20 # Save knowledge to file
21 def save_knowledge(knowledge_base):
22     with open(KNOWLEDGE_BASE_FILE, "w") as file:
23         json.dump(knowledge_base, file, indent=4)
24     print("[✓] Knowledge saved to", KNOWLEDGE_BASE_FILE)
25 
26 # Get or teach a response
27 def get_response(user_input, knowledge_base):
28     user_input = user_input.lower()
29     if user_input in knowledge_base:
30         return knowledge_base[user_input]
31     else:
32         response = input(f"I don't know how to respond to '{user_input}'. How should I reply? ")
33         knowledge_base[user_input] = response
34         save_knowledge(knowledge_base)
35         return response
36 
37 # Main chatbot loop
38 def chat():
39     print("🤖 Chatbot started. Type 'exit' to quit.")
40     knowledge_base = load_knowledge()
41     while True:
42         user_input = input("You: ")
43         if user_input.lower() == "exit":
44             print("🤖 Goodbye!")
45             break
46         reply = get_response(user_input, knowledge_base)
47         print(f"Bot: {reply}")
48 
49 if __name__ == "__main__":
50     chat()
